305  struct ImageIndexHash
306  {
312  std::size_t operator()(const ImageIndex& value) const
313  {
314  const std::size_t combineFactor = 0x9e3779b9;
315  std::hash<unsigned int> hasher;
316  size_t curHash = hasher(static_cast<unsigned int>(value.cubeFace));
317  curHash ^= hasher(value.mipLevel) + combineFactor + (curHash << 6) + (curHash >> 2);
318  curHash ^= hasher(value.depth) + combineFactor + (curHash << 6) + (curHash >> 2);
319  return curHash;
320  }
